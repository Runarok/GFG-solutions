<pre><span style="font-size: 18px;"><strong>Input :</strong> arr[] = [10, 9, 4, 5, 4, 8, 6]

</span></pre>
<pre><span style="font-size: 18px;"><strong>Input :</strong> arr[] = [1, 2, 3, 2, 3, 7, 2, 1]

</span></pre>
